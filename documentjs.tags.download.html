@download

Adds a download link.

###Example:

/**
 * @constructor jQuery.Drag
 * ...
 * @plugin jquery/event/drag
 * @download jquery/dist/jquery.event.drag.js
 * @test jquery/event/drag/qunit.html
 * ...
 */
 $.Drag = function(){}
