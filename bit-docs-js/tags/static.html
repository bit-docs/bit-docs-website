@static module

Declares that @property and @function tags belong to the preceeding @constructor.

@prototype 

/**
 * @constructor
 * Creates an Animal
 */
Animal = function(){ ... }
/** @prototype */
Animal.prototype = {
   /**
    * Eats another animal.
    */
    eat: function(animal){ ... }
}
